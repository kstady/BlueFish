\section*{Dividing Fractions}


\begin{definition}[\textbf{Reciprocal of a Fraction}]

The \textbf{reciprocal} of the fraction $\displaystyle \frac{a}{b}$ is $\displaystyle \frac{b}{a}.$
\end{definition}


\begin{remark}

\begin{enumerate}
\item  The reciprocal of a fraction is the fraction with the numerator and
denominator interchanged (inverted).

\item The \textbf{reciprocal of an integer} is found by writing the integer over 1 and then interchanging 
the numerator and denominator. For example, to find the reciprocal of $-4,$ write it as $-\frac{4}{1},$ t
hen interchange the numerator and denominator to get $-\frac{1}{4}.$

\item Two numbers are reciprocals of each other if their product is 1.
\[ \frac{a}{b} \cdot \frac{b}{a} = 1 \]

\item Every number has a reciprocal except 0.  This is because if we write 0 as $\frac{0}{1}$
and then take the reciprocal we get $\frac{1}{0}$ which is undefined.  Furthermore, if 0 had a reciprocal
then the product of 0 and its reciprocal should equal 1, but we know that the product of 0 and any number equals 0.
\end{enumerate}

\end{remark}

\begin{definition}[\textbf{Dividing Fractions}]


If $a, b, c$ and $d$ are numbers and $b, c$ and $d$ are not 0, then
\[ \frac{a}{b} \div \frac{c}{d} = \frac{a}{b} \cdot \frac{d}{c} \]

To divide with fractions, multiply the first fraction by the reciprocal of the second fraction.

\end{definition}

\begin{warning}  
    
DO NOT SIMPLIFY until you have written the division problem as an equivalent multiplication problem.  In the division problem below, 
you might think that the 5s cancel, however by rewriting it as multiplication, you can see that the 5s do not
cancel.

\[ \frac{2}{5} \div \frac{5}{3}  = \frac{2}{5} \cdot \frac{3}{5} \]

\end{warning}

\begin{problem}
Divide. Remember to simplify your answer, if possible.

\[  \frac{10}{11} \div \frac{4}{5} = \frac{\answer{25}}{\answer{22}} \]

\[ 7 \div \frac{42}{3} = \frac{\answer{1}}{\answer{2}} \]

\[ \frac{9}{8} \div \frac{3}{7} = \frac{\answer{21}}{\answer{8}} \]

\end{problem}

\begin{problem}
You have a ribbon that is $\frac{5}{6}$ of a yard long.  You need to cut the ribbon into shorter pieces, each 
measuring exactly $\frac{1}{12}$ of a yard.  How many pieces can you cut from the ribbon?  Begin by setting up  
the division problem, then find the answer.  

\begin{multipleChoice}
\choice{$\frac{5}{6} \div 12$}
\choice[correct]{$\frac{5}{6} \div \frac{1}{12}$}
\choice{$\frac{1}{12} \div \frac{5}{6}$}
\choice{$12 \div \frac{5}{6}$}
\end{multipleChoice}

$\answer{10}$ \ pieces

\begin{hint}
Divide the total length by the length of each piece.
\end{hint}

\end{problem}
